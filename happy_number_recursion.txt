int square(int);
void calculation(int);

{

int square(int q) {
	int res;
	res = q * q;
	check++;

	}

void calculation(int b) {

	int temp, arr[100], i{}, sum = 0;
	
	while (b != 0)
		{
		temp = b;
		b = b % 10;
		arr[i] = b;
		sum = sum + (arr[i] * arr[i]);
		i++;
		b = temp / 10;
		} 
	 check++;

	if (sum == 1)
	{
		cout << "Number is happy\n";
		cout << check << endl;
	}
	else 		
			calculation(sum);
}
